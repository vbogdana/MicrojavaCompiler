public void stringLiteral(String s) {
		Code.loadConst(s.length() - 2); // n
		Code.put(Code.newarray);
		Code.put(0);
		
		int i = 0;
		for (; i < s.length() - 2; i++) {
			Code.put(Code.dup);
			Code.loadConst(i);
			Code.loadConst(s.charAt(i+1));
			Code.put(Code.bastore);
		}
	}
